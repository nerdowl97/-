function openCharacterProfile(charId) { const proj = getActiveProject(); const char = proj.characters.find(c => c.id === charId); if (!char) return; // Update app views to render custom details activeView = 'character-profile'; // Show view container and hide companions const views = [ 'dashboard', 'synopsis', 'editor', 'characters', 'relations', 'timeline', 'locations', 'items', 'quotes', 'laws', 'moodboard', 'consistency', 'ai-assistant' ]; views.forEach(v => { const el = document.getElementById(`view-${v}`); if (el) el.classList.add('hidden'); }); document.getElementById('view-character-profile').classList.remove('hidden'); // Wire top-level actions document.getElementById('profile-btn-edit').onclick = () => openEditCharacterModal(charId); document.getElementById('profile-btn-delete').onclick = () => { if (confirm(`هل أنت متأكد من رغبتك في مسح ملف الشخصية "${char.name}" بالكامل؟`)) { deleteCharacter(charId); switchView('characters'); } }; document.getElementById('profile-btn-print').onclick = () => window.print(); document.getElementById('profile-btn-share').onclick = () => { const link = `[شخصية: ${char.name}]`; const dummy = document.createElement("input"); document.body.appendChild(dummy); dummy.value = link; dummy.select(); document.execCommand("copy"); document.body.removeChild(dummy); alert(`تم نسخ مرجع الشخصية الذكي للذاكرة: ${link}\nيمكنك الآن لصقه مباشرة في محرر نصوص الفصول.`); };
